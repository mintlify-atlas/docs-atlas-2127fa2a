        ```python
        # spacy: named entity recognition
        import spacy

        nlp = spacy.load("en_core_web_sm")
        doc = nlp("Apple is looking at buying U.K. startup for $1 billion.")

        for ent in doc.ents:
            print(ent.text, ent.label_)

        # U.K.   GPE
        # $1 billion  MONEY
        ```

        ```python
        # gensim: train a Word2Vec model
        from gensim.models import Word2Vec

        sentences = [
            ["machine", "learning", "is", "fun"],
            ["deep", "learning", "powers", "modern", "ai"],
            ["python", "is", "great", "for", "machine", "learning"],
        ]

        model = Word2Vec(sentences, vector_size=100, window=5, min_count=1, workers=4)
        similar = model.wv.most_similar("learning", topn=3)
        print(similar)
        ```

    ```python
    # scikit-surprise: train and evaluate a collaborative filtering model
    from surprise import SVD, Dataset, accuracy
    from surprise.model_selection import train_test_split

    data = Dataset.load_builtin("ml-100k")
    trainset, testset = train_test_split(data, test_size=0.25, random_state=42)

    model = SVD(n_factors=100, n_epochs=20)
    model.fit(trainset)
    predictions = model.test(testset)

    accuracy.rmse(predictions)
    ```

    ```python
    # annoy: build an index and query nearest neighbors
    from annoy import AnnoyIndex
    import random

    dims = 128
    index = AnnoyIndex(dims, "angular")

    for i in range(1000):
        vector = [random.gauss(0, 1) for _ in range(dims)]
        index.add_item(i, vector)

    index.build(n_trees=10)
    index.save("embeddings.ann")

    # Query
    query = [random.gauss(0, 1) for _ in range(dims)]
    neighbors = index.get_nns_by_vector(query, n=5)
    print("Nearest neighbors:", neighbors)
    ```
