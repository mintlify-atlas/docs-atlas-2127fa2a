    ```python
    # langchain: build a chain using LCEL (LangChain Expression Language)
    from langchain_openai import ChatOpenAI
    from langchain_core.prompts import PromptTemplate
    from langchain_core.output_parsers import StrOutputParser

    llm = ChatOpenAI(model="gpt-4o")
    prompt = PromptTemplate.from_template("Explain {topic} in simple terms.")
    chain = prompt | llm | StrOutputParser()
    result = chain.invoke({"topic": "quantum computing"})
    print(result)
    ```

    ```python
    # pydantic-ai: type-safe agent with structured output
    from pydantic_ai import Agent

    agent = Agent('openai:gpt-4o', system_prompt='Be a helpful assistant.')
    result = agent.run_sync('What is the capital of France?')
    print(result.data)
    ```

    ```python
    # instructor: extract structured data from an LLM
    import instructor
    from openai import OpenAI
    from pydantic import BaseModel

    client = instructor.from_openai(OpenAI())

    class UserDetail(BaseModel):
        name: str
        age: int

    user = client.chat.completions.create(
        model="gpt-4o",
        response_model=UserDetail,
        messages=[{"role": "user", "content": "Extract: John is 30 years old."}],
    )
    print(user.name, user.age)
    ```
