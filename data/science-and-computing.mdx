```python numpy
import numpy as np

# Matrix operations
A = np.array([[1, 2], [3, 4]])
eigenvalues, eigenvectors = np.linalg.eig(A)
print("Eigenvalues:", eigenvalues)
```

```python scipy
from scipy import stats

data = [2.1, 2.5, 2.3, 2.8, 2.4, 2.7]
t_stat, p_value = stats.ttest_1samp(data, popmean=2.5)
print(f"t-statistic: {t_stat:.3f}, p-value: {p_value:.3f}")
```
